\section{Geometric Stacks à la Toën-Vezzosi}

\begin{definition}[Homotopical Algebraic Context]
    A homotopical algebraic context is the data of a triple $(\cat{C}, \cat{C}_0, \A)$ where $\cat{C}$ is a symmetric-monoïdal presentable $\infty$-category with full sub-categories $\cat{C}_0 \mono \cat{C}$ and $\cat{A} \mono \EAlg{\infty}(\cat{C})$ such that the assumptions ($1.1.0.n$) are satisfied:
    \begin{enumerate}
        \item[$(1)$] $\cat{C}$ is $0$-semiadditive (zero object, finite direct sums)
        \item[$(2,3,4)$] \textbf{void}
        \item[$(6)$] $\cat{C}_0$ is closed under colimits and tensor products (it's meant to be the connective objects)
        \item[$(\mathbf{good})$] for $A \in \A$, the full sub-category $A\amod_{\cat{C}_0} \mono A\amod_\cat{C}$ is dense
    \end{enumerate}
\end{definition}

For our purpose: $\cat{C} = \cat{C}_0 = \mathbb{R}\amod^{\geq 0}$, $\A = \EAlg{\infty}(\mathbb{R}\amod^{\geq 0})$.

\begin{definition}[HAG Context]
    A homotopical algebraic geometry context is a quintuple $(\cat{C}, \cat{C}_0, \A, \tau, \mathbf{P})$ where $(\cat{C}, \cat{C}_0, \A)$ is a homotopical algebraic context, $\tau$ is a pre-topology on $\aff_{\cat{C}} = \EAlg{\infty}(\cat{C})^\op$ and $\mathbf{P}$ is a class of morphisms in $\aff_{\cat{C}}$ satisfying ($1.3.2.n$)
    \begin{enumerate}
        \item[$(2)$]
        \begin{enumerate}
            \item $\tau$ is \textbf{quasi-compact}: existence of finite sub-covering families,
            \item contains coproduct diagrams,
            \item \textbf{hypercomplete}?
        \end{enumerate}
        
        \item[$(11)$]
        \begin{enumerate}
            \item morphisms in $\tau$ are in $\mathbf{P}$,
            \item $\mathbf{P}$ is stable under composition and pullbacks,
            \item being in $\mathbf{P}$ can be checked $\tau$-locally on the domain,
            \item projections into finite coproducts are in $\mathbf{P}$ (unecessary with the $\tau$-assumption $(b)$)
        \end{enumerate}
    \end{enumerate}
\end{definition}

\subsection{Coaffine Stacks}

We define $\Spec\in \A \to \st$ as the Yoneda embedding. We can then extend this functor along $\cat{A} \mono \EAlg{\infty}(\cat{C})$ to get $\Spec\in \EAlg{\infty}(\cat{C}) \to \pst$ defined by:
\[
    \Spec(A)(B) = \EAlg{\infty}(\cat{C})(B,A)
\]

\begin{remark}[Lack of coDensity]
    $\Spec\in \EAlg{\infty}(\cat{C}) \to \st$ being fully-faithful is equivalent to \textbf{codensity} of $\cat{A} \mono \EAlg{\infty}(\cat{C})$. This is \textbf{generally false} (all $A \in \EAlg{\infty}(k\amod)$ with multiplicative units in non-zero degree are mapped to the terminal prestack).
\end{remark}

In Lurie [DAG 8, 4.4], it is shown however that this functor is fully faithful in the ``coconnective'' case.

\begin{definition}[coconnective]
    An algebra $A \in \EAlg{n}(k\amod)$ is called \textbf{coconnective} if its unit $k \to A$ is a connective cover. They form a category $\EAlg{n}(k\amod)^{cc}$.
\end{definition}

\begin{theorem}[Lurie, DAG 8, 4.4] On a field of characteristic zero, the restricted functor $\Spec\in \EAlg{\infty}(k\amod)^{cc} \mono \pst$ is fully-faithful. Furthermore, the functor $\EAlg{\infty}(k\amod)^{cc} \mono [\EAlg{\infty}(k\amod^\heartsuit) , \Spa]$ is also fully-faithful and its image lies in the image of cosimplicial algebras (its a totalization of discrete algebras).
\end{theorem}
