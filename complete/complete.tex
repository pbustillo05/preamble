\section{Introduction}

Here is (up to potential mistakes) how the usual definition of the category of cobordisms goes:

\begin{definition}[Pre-Bordism Category]
    We define an $(n+1)$-fold simplicial set $\PBord^n$ as follows.
    The set $\PBord^n_{\mathbf{i},j}$ is given by the set of
    \begin{enumerate}
        \item Smooth maps $\left((a^l_k, b^l_k \inset \Delta^j_e \to \mathbb{R})_{0 \leq k \leq i_l}\right)_{1 \leq l \leq n}$ such that
        \[
            \begin{cases}
                a^l_k < b^l_k \\
                a^l_{k-1} \leq a^l_k \\
                b^l_{k-1} \leq b^l_k 
            \end{cases}
        \]
        \item $M$ compact $(n+j)$-dimensional submanifold of $\mathbb{R}^\infty \times \left(\prod_{1 \leq l \leq n} (a^l_0, b^l_{i_l})\right) \underline{\times} \Delta^j_e$ such that:
        \begin{enumerate}
            \item \textbf{Family:} the composition $\pi \in M \mono \mathbb{R}^\infty \times \left(\prod_{1 \leq l \leq n} (a^l_0, b^l_{i_l})\right) \underline{\times} \Delta^j_e \epi \left(\prod_{1 \leq l \leq n} (a^l_0, b^l_{i_l})\right) \underline{\times} \Delta^j_e$ is proper,
                and the further projection to $\Delta^j_e$ is a submersion which is trivial on the extended part of the simplex,
            \item \textbf{Reducedness:} for every $1 \leq d\leq n$ and every $0 \leq k \leq i_d$, $M_{\mid} \epi \left( (a^d_k, b^d_k) \times \prod_{d \leq l \leq n} (a^l_0, b^l_{i_l}) \right) \underline{\times} \Delta^j_e$ is a submersion,
        \end{enumerate}
    \end{enumerate}
\end{definition}

Famously, this is not univalent.
The goal of this paper is to construct its univalent completion explicitly.
The main intuition is that making this category univalent involves seeing a $k$-morphism not as a $k$-manifold 

\subsection{Tameness}

\begin{definition}[End Space]
    Let $W$ be a topologial space.
    The \textbf{end space} $e(W)$ of $W$ is the space of \textbf{proper paths} $[0, \infty) \to W$.
\end{definition}

\begin{definition}[Ends]
    Let $W$ be a topologial space.
    A neighbourhood of infinity in $W$ is an open subset $U \subset W$ that contains the complenent of a compact.
    An \textbf{end} is an equivalence class of decreasing sequences of neighbourhood of infinity $\ldots \subset U_n \subset \ldots \subset U_1 \subset W$ such that $\bigcap_{n \geq 1} \overline{U_n} = \emptyset$.
    They're subject to the obvious equivalence relation.
\end{definition}

\begin{definition}[Tame Ends]
    And end is \textbf{tame} if the inclusions $U_i \mono U_j$ are homotopy equivalences for all $i < j$.
\end{definition}

\subsection{Control Topology in Families}

\begin{definition}[Controlled Famillies, I]
    Let $X$ be a smooth manifold and $I$ a compact manifold with boundary.
    Let $W \subset \mathbb{R}^\infty \times I^o \times X$ be a submanifold such that
    \begin{enumerate}
        \item $W \to X$ is a smooth fiber bundle,
        \item for all $t \inset X$, $W_t \epi I$ is a proper submersion,
    \end{enumerate}
    We say that $W \to I \times X$ is \textbf{controlled} in the $I$-direction if
        it extends to a topological subspace $\overline{W} \subset \mathbb{R}^\infty \times I \times X$ such that
            $\overline{W}^o = W$ and
            for all $t \inset X$, the projection $\partial \overline{W}_t \simeq \partial I$ is a homeomorphism.
\end{definition}

Firslty, it should be equivalent to $O(1)$-variance. \pablo{formulate a precise result}
Secondly, for sufficiently high dimensions, this should become an algebraic condition.

\subsection{Univalence is Control}

\begin{definition}
    We define an $(n+1)$-fold simplicial set $\Bord^n$ as follows.
    The set $\Bord^n_{\mathbf{i},j}$ is given by the set of
    \begin{enumerate}
        \item Continuous maps $\left((a^l_k, b^l_k \inset \Delta^j_e \to \mathbb{R})_{0 \leq k \leq i_l}\right)_{1 \leq l \leq n}$ such that
        \[
            \begin{cases}
                a^l_k < b^l_k \\
                a^l_{k-1} \leq a^l_k \\
                b^l_{k-1} \leq b^l_k 
            \end{cases}
        \]
        \item $M$ compact $(n+j)$-dimensional submanifold of $\mathbb{R}^\infty \times \left(\prod_{1 \leq l \leq n} (a^l_0, b^l_{i_l})\right) \underline{\times} \Delta^j_e$ such that:
        \begin{enumerate}
            \item \textbf{Family:} the composition $\pi \in M \mono \mathbb{R}^\infty \times \left(\prod_{1 \leq l \leq n} (a^l_0, b^l_{i_l})\right) \underline{\times} \Delta^j_e \epi \Delta^j_e$ is a smooth fiber bundle,
            \item \textbf{Reducedness:} for every $t \inset \Delta^j_e$, $1 \leq d\leq n$ and every $0 \leq k \leq i_d$, $M_t \epi (a^d_k(t), b^d_k(t)) \times \prod_{d \leq l \leq n} (a^l_0(t), b^l_{i_l}(t))$ is a proper submersion of codimension $d-1$,
            \item \textbf{Control:} the $t$-families above of proper submersions are controlled in the categorical directions.
        \end{enumerate}
    \end{enumerate}
\end{definition}

\begin{remark}[Looping]
    One can replace $(b)$ by ``proper submersion of codimension $d$'' and this becomes $\Omega^d \Bord^n$ where objects are compact $d$ dimensional manifolds.

\subsection{Invertible Data as a Paving}

For $n \geq 0$, we have a distance on $\Bin{n}$ given by the cyclic picture:
    $d(i,j) = \min\{|i-j|, 2^n - |i-j|\}$ for all $i,j \inset \Bin{n}$.
This is the shortest path distance for the undirected graph given by the edges $(i, i+1)$ for all $i \inset \Bin{n}$.
This induces a product distance on $\Tau{n} \coloneq \Bin{n} \times \Bin{n+1}$.

\begin{definition}[Exclusion on the Modulo Torus]
    For $n \geq 0$, we define a subset $E_n \subset \Tau{n}$ as
    \[
        E_n = \{(i,j) \inset \Tau{n} \mid (i,j) \textnormal{ is at distance } \geq 2^{n-1} \textnormal{ from both } (0,0), (2^{n-1},2^n)\}
    \]
\end{definition}

\begin{definition}[Nodes of the Paving]
    For $n \geq 0$, we define a subset $A_n \subset \mathbb{Z}^n$ as
    \[
        A_n = \{i \inset \mathbb{Z}^n \mid \forall 0 \leq j \leq n-1, \; (i_j, i_{j+1}) \inset E_n\}
    \]
\end{definition}

\begin{definition}[Paving]
    A rectangle in $\mathbb{Z}^n$ is a choice of a set of corners $R = \prod_{i = 1}^n R_i \subset \mathbb{Z}^n$ where $R_i \subset \mathbb{Z}$ is of cardinality either $1$ or $2$.
    A paving is a collection of rectangles forming a cell decomposition of $\mathbb{R}^n$.
    For $n \geq 0$, we have a paving $\mathcal{A}_n$ of $\mathbb{R}^n$ given by the following rule:
    \begin{enumerate}
        \item for all rectangles $R$, we have $R \inset \mathcal{A}_n$ if and only if $R \subset A_n$ and $\Hull(R)^\circ \cap A_n = \emptyset$.
    \end{enumerate}
\end{definition}

    What we're doing is we're providing the ways to "fill" the gabs to create a diffeomorphism.
